// services/PayrollAPI.js
import axios from 'axios';

const API_BASE_URL = process.env.REACT_APP_API_URL || 'http://localhost:3001/api';

class PayrollAPI {
  
  // Configure axios defaults
  static setupInterceptors() {
    axios.defaults.baseURL = API_BASE_URL;
    axios.defaults.headers.common['Authorization'] = `Bearer ${localStorage.getItem('token')}`;
    
    // Request interceptor
    axios.interceptors.request.use(
      (config) => {
        const token = localStorage.getItem('token');
        if (token) {
          config.headers.Authorization = `Bearer ${token}`;
        }
        return config;
      },
      (error) => Promise.reject(error)
    );
    
    // Response interceptor for error handling
    axios.interceptors.response.use(
      (response) => response,
      (error) => {
        if (error.response?.status === 401) {
          localStorage.removeItem('token');
          window.location.href = '/login';
        }
        return Promise.reject(error);
      }
    );
  }
  
  // Payroll Periods
  static async getPeriods(params = {}) {
    const response = await axios.get('/payroll/periods', { params });
    return response.data;
  }
  
  static async getPeriodDetails(periodId) {
    const response = await axios.get(`/payroll/periods/${periodId}`);
    return response.data;

    const response = await axios.get(`/payroll/staff/${staffId}/allowances`);
    return response.data;
  }
  
  static async assignAllowance(staffId, data) {
    const response = await axios.post(`/payroll/staff/${staffId}/allowances`, data);
    return response.data;
  }
  
  static async removeAllowance(staffId, allowanceId) {
    const response = await axios.delete(`/payroll/staff/${staffId}/allowances/${allowanceId}`);
    return response.data;
  }
  
  static async getStaffDeductions(staffId) {
    const response = await axios.get(`/payroll/staff/${staffId}/deductions`);

  }

// components/StaffPayrollManagement.jsx
import React, { useState, useEffect } from 'react';

  Table, Card, Button, Modal, Form, Select, InputNumber, 
  DatePicker, Tag, Space, notification, Popconfirm
} from 'antd';
import { 
  PlusOutlined, 
  EditOutlined, 
  DeleteOutlined, 
  UserAddOutlined 
} from '@ant-design/icons';

const { Option } = Select;

const StaffPayrollManagement = () => {
  const [loading, setLoading] = useState(false);
  const [staff, setStaff] = useState([]);
  const [selectedStaff, setSelectedStaff] = useState(null);
  const [modalVisible, setModalVisible] = useState(false);
  const [modalType, setModalType] = useState(''); // 'enroll', 'suspend', 'promote', 'allowance', 'deduction'
  const [allowanceTypes, setAllowanceTypes] = useState([]);
  const [deductionTypes, setDeductionTypes] = useState([]);
  const [gradeLevels, setGradeLevels] = useState([]);
  const [gradeSteps, setGradeSteps] = useState([]);
  const [form] = Form.useForm();

  useEffect(() => {
    loadStaff();
    loadAllowanceTypes();
    loadDeductionTypes();
    loadGradeLevels();
  }, []);

  const loadStaff = async () => {
    setLoading(true);
    try {
      const response = await PayrollAPI.getEnrolledStaff();
      setStaff(response.data);
    } catch (error) {
      console.error('Error loading staff:', error);
      notification.error({
        message: 'Error',
        description: 'Failed to load staff data'
      });
    } finally {
      setLoading(false);
    }
  };

  const loadAllowanceTypes = async () => {
    try {
      const response = await PayrollAPI.getAllowanceTypes();
      setAllowanceTypes(response.data);
    } catch (error) {
      console.error('Error loading allowance types:', error);
    }
  };

  const loadDeductionTypes = async () => {
    try {
      const response = await PayrollAPI.getDeductionTypes();
      setDeductionTypes(response.data);
    } catch (error) {
      console.error('Error loading deduction types:', error);
    }
  };

  const loadGradeLevels = async () => {
    try {
      const response = await PayrollAPI.getGradeLevels();
      setGradeLevels(response.data);
    } catch (error) {
      console.error('Error loading grade levels:', error);
    }
  };

  const loadGradeSteps = async (gradeId) => {
    try {
      const response = await PayrollAPI.getGradeSteps(gradeId);
      setGradeSteps(response.data);
    } catch (error) {
      console.error('Error loading grade steps:', error);
    }
  };

  const handleEnrollStaff = (record) => {
    setSelectedStaff(record);
    setModalType('enroll');
    setModalVisible(true);
    form.resetFields();
  };

  const handleSuspendStaff = async (record) => {
    try {
      await PayrollAPI.suspendStaff(record.staff_id, {
        reason: 'Suspended from payroll',
        effective_date: dayjs()
      });
      
      notification.success({
        message: 'Success',
        description: 'Staff suspended from payroll successfully'
      });
      
      loadStaff();
    } catch (error) {
      notification.error({
        message: 'Error',
        description: 'Failed to suspend staff'
      });
    }
  };

  const handlePromoteStaff = (record) => {
    setSelectedStaff(record);
    setModalType('promote');
    setModalVisible(true);
    form.setFieldsValue({
      current_grade_id: record.grade_id,
      current_step_id: record.step_id
    });
  };

  const handleAssignAllowance = (record) => {
    setSelectedStaff(record);
    setModalType('allowance');
    setModalVisible(true);
    form.resetFields();
  };

  const handleAssignDeduction = (record) => {
    setSelectedStaff(record);
    setModalType('deduction');
    setModalVisible(true);
    form.resetFields();
  };

  const handleModalSubmit = async (values) => {
    try {
      setLoading(true);
      
      switch (modalType) {
        case 'enroll':
          await PayrollAPI.enrollStaff(selectedStaff.staff_id, values);
          notification.success({
            message: 'Success',
            description: 'Staff enrolled in payroll successfully'
          });
          break;
          
        case 'promote':
          await PayrollAPI.promoteStaff(selectedStaff.staff_id, values);
          notification.success({
            message: 'Success',
            description: 'Staff promoted successfully'
          });
          break;
          
        case 'allowance':
          await PayrollAPI.assignAllowance(selectedStaff.staff_id, values);
          notification.success({
            message: 'Success',
            description: 'Allowance assigned successfully'
          });
          break;
          
        case 'deduction':
          await PayrollAPI.assignDeduction(selectedStaff.staff_id, values);
          notification.success({
            message: 'Success',
            description: 'Deduction assigned successfully'
          });
          break;
      }
      
      setModalVisible(false);
      loadStaff();
      
    } catch (error) {
      notification.error({
        message: 'Error',
        description: error.response?.data?.message || 'Operation failed'
      });
    } finally {
      setLoading(false);
    }

  const [loans, setLoans] = useState([]);
  const [staff, setStaff] = useState([]);
  const [modalVisible, setModalVisible] = useState(false);
  const [detailModalVisible, setDetailModalVisible] = useState(false);

  useEffect(() => {
    loadLoans();
    loadStaff();
  }, []);

  const loadLoans = async () => {
    setLoading(true);
    try {
      const response = await PayrollAPI.getLoans();
      setLoans(response.data);
    } catch (error) {
      console.error('Error loading loans:', error);
      notification.error({
        message: 'Error',
        description: 'Failed to load loans data'
      });
    } finally {
      setLoading(false);
    }
  };

  const loadStaff = async () => {
    try {
      const response = await PayrollAPI.getEnrolledStaff();
      setStaff(response.data);
    } catch (error) {
      console.error('Error loading staff:', error);
    }
  };

  const handleCreateLoan = async (values) => {
    try {
      setLoading(true);
      
      // Calculate loan details
      const principal = values.principal_amount;
      const interestRate = values.interest_rate || 0;
      const months = values.loan_duration_months;
      
      const totalAmount = principal + (principal * interestRate / 100);
      const monthlyDeduction = totalAmount / months;
      
      const loanData = {
        ...values,
        total_amount: totalAmount,
        monthly_deduction: monthlyDeduction,
        balance_remaining: totalAmount,
        expected_end_date: moment().add(months, 'months').format('YYYY-MM-DD')
      };
      
      await PayrollAPI.createLoan(loanData);
      
      notification.success({
        message: 'Success',
        description: 'Loan created successfully'
      });
      
      setModalVisible(false);
      form.resetFields();
      loadLoans();
      
    } catch (error) {
      notification.error({
        message: 'Error',
        description: error.response?.data?.message || 'Failed to create loan'
      });
    } finally {
      setLoading(false);
    }
  };

  const handleApproveLoan = async (loanId) => {
    try {
      await PayrollAPI.approveLoan(loanId, {
        approval_notes: 'Loan approved for payroll deduction'
      });
      
      notification.success({
        message: 'Success',
        description: 'Loan approved successfully'
      });
      
      loadLoans();
      
    } catch (error) {
      notification.error({
        message: 'Error',
        description: 'Failed to approve loan'
      });
    }
  };

  const showLoanDetails = (loan) => {
    setSelectedLoan(loan);
    setDetailModalVisible(true);
  };

  const calculateProgress = (loan) => {
    const paid = loan.total_amount - loan.balance_remaining;
    return (paid / loan.total_amount) * 100;
  };

  const columns = [
    {
      title: 'Loan ID',
      dataIndex: 'loan_id',
      key: 'loan_id',
      width: 80
    },

      title: 'Staff Name',
      dataIndex: ['staff', 'name'],
      key: 'staff_name',
      width: 200
    },
    {
      title: 'Loan Type',
      dataIndex: 'loan_type',
      key: 'loan_type',
      width: 120
    },
    {
      title: 'Principal',
      dataIndex: 'principal_amount',
      key: 'principal_amount',
      width: 120,
      render: (amount) => `₦${Number(amount).toLocaleString()}`
    },
    {
      title: 'Total Amount',
      dataIndex: 'total_amount',
      key: 'total_amount',
      width: 120,
      render: (amount) => `₦${Number(amount).toLocaleString()}`
    },
    {
      title: 'Monthly Deduction',
      dataIndex: 'monthly_deduction',
      key: 'monthly_deduction',
      width: 130,
      render: (amount) => `₦${Number(amount).toLocaleString()}`
    },
    {
      title: 'Balance',
      dataIndex: 'balance_remaining',
      key: 'balance_remaining',
      width: 120,
      render: (amount) => `₦${Number(amount).toLocaleString()}`
    },
    {
      title: 'Progress',
      key: 'progress',
      width: 120,
      render: (_, record) => (
        <Progress 
          percent={Math.round(calculateProgress(record))} 
          size="small"
          status={record.status === 'completed' ? 'success' : 'active'}
        />
      )
    },
    {
      title: 'Status',
      dataIndex: 'status',
      key: 'status',
      width: 100,
      render: (status) => {
        const colors = {
          active: 'blue',
          completed: 'green',
          suspended: 'red'
        };
        return (
          <Tag color={colors[status]}>
            {status.toUpperCase()}
          </Tag>
        );
      }
    },
    {
      title: 'Actions',
      key: 'actions',
      width: 150,
      render: (_, record) => (
        <Space size="small">
          <Button
            type="link"
            icon={<EyeOutlined />}
            onClick={() => showLoanDetails(record)}
          >
            View
          </Button>
          {record.status === 'pending' && (

  ];

  const loanStats = {
    totalLoans: loans.length,
    activeLoans: loans.filter(l => l.status === 'active').length,

    totalPrincipal: loans.reduce((sum, l) => sum + Number(l.principal_amount), 0),
    totalOutstanding: loans.filter(l => l.status === 'active')
                          .reduce((sum, l) => sum + Number(l.balance_remaining), 0)
  };

  return (
    <div className="loan-management">
      {/* Statistics Cards */}
      <Row gutter={16} style={{ marginBottom: 24 }}>
        <Col span={6}>

        />
      </Card>

      {/* Create Loan Modal */}
      <Modal
